class F_COLORS:
    BLUE = '\033[94m'
    WARNING = '\033[93m'
    FAIL = '\033[91m'
    NORMAL = '\033[0m'
    BOLD = '\033[1m'
    UNDERLINE = '\033[4m'
    TEST = '\33[6m'
    
class B_COLORS:
    BLUE = '\33[44m'
    WARNING = '\33[43m'
    FAIL = '\33[101m'
    NORMAL = '\033[0m'
    
        

print('pythonのprintには、色をつけることができる')
print(F_COLORS.BLUE + "-------OK-------" + F_COLORS.NORMAL)
print(F_COLORS.WARNING + "-------WARNING-------" + F_COLORS.NORMAL)
print(F_COLORS.FAIL + "-------FAILED-------" + F_COLORS.NORMAL)
